new内存初始化 

#include <iostream>

using namespace std;

int main()
{
    int *arrInu = new int[1024];
    for(int idx = 0; idx < 1024; idx++){
        arrInu[idx] = idx;
    }
    delete []arrInu;
    arrInu = nullptr;

    //数组不会初始化为0
    int *arrIno = new int[1024];
    for(int idx = 0; idx < 1024; idx++){
        cout << arrIno[idx] << " ";
    }
    cout << endl;
    delete []arrIno;
    arrIno = nullptr;

    //数组会初始化为0
    int *arrIno0 = new int[1024]();
    for(int idx = 0; idx < 1024; idx++){
        cout << arrIno0[idx] << " ";
    }
    cout << endl;
    delete []arrIno0;
    arrIno0 = nullptr;  

    return 0;
}
